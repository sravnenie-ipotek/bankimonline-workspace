Financial Calculation Test Example

            

                
describe('Property Ownership LTV Logic', () => {
  it('should apply 75% LTV for no property ownership', () => {
    const result = calculateMaxLoan(1000000, 'no_property');
    expect(result.maxLoanAmount).toBe(750000);
    expect(result.minDownPayment).toBe(250000);
    expect(result.ltvRatio).toBeCloseTo(0.75, 3);
  });

  it('should handle floating-point precision correctly', () => {
    const result = calculateMonthlyPayment(750000, 0.035, 25);
    expect(result).toBeCloseTo(3756.85, 2); // Verified manually
  });

  it('should handle mathematical edge cases', () => {
    expect(() => calculateMaxLoan(0, 'no_property')).not.toThrow();
    expect(() => calculateMaxLoan(-100000, 'no_property')).toThrow();
    expect(calculateMaxLoan(Infinity, 'no_property')).toBe(Infinity);
  });
});

            


            
React Component Test Example

            

                
import { render, screen } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { axe, toHaveNoViolations } from 'jest-axe';

expect.extend(toHaveNoViolations);

describe('MortgageCalculatorStep1', () => {
  const user = userEvent.setup();
  
  beforeEach(() => {
    jest.clearAllMocks();
    server.resetHandlers();
  });

  it('should render Hebrew text with RTL support', () => {
    render(<MortgageCalculatorStep1 language="he" />);
    expect(document.dir).toBe('rtl');
    expect(screen.getByLabelText(/ערך נכס/)).toBeInTheDocument();
  });

  it('should not have accessibility violations', async () => {
    const { container } = render(<MortgageCalculatorStep1 />);
    const results = await axe(container);
    expect(results).toHaveNoViolations();
  });

  it('should validate property ownership selection', async () => {
    render(<MortgageCalculatorStep1 />);
    
    await user.selectOptions(
      screen.getByLabelText(/property ownership/i),
      'no_property'
    );
    
    await user.click(screen.getByRole('button', { name: /continue/i }));
    
    expect(screen.getByText(/75% financing/i)).toBeInTheDocument();
  });
});
